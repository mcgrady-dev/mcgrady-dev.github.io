    ```kotlin
      public static int fun(int n) {
        if (n <= 1) {
          return 1;
        } else {
          return fun(n - 1) * n;
        }
      }
      
      public static void main(String[] args) {
        System.out.println("3的阶乘是：" + fun(3));
      }
    }
    ```
